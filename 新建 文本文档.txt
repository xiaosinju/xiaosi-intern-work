#!/bin/bash

# 设置变量
GITHUB_USER="your-username"
GITHUB_REPO="your-repo"
REMOTE_FILE_PATH="path/in/repo/file.txt"
LOCAL_FILE_PATH="/path/to/local/file.txt"

# GitHub 文件的原始 URL
GITHUB_RAW_URL="https://raw.githubusercontent.com/${GITHUB_USER}/${GITHUB_REPO}/main/${REMOTE_FILE_PATH}"

# 下载文件并覆盖本地文件
curl -o ${LOCAL_FILE_PATH} ${GITHUB_RAW_URL}

# 检测下载是否成功
if [ $? -eq 0 ]; then
  echo "File downloaded and saved to ${LOCAL_FILE_PATH}"
else
  echo "Failed to download file from ${GITHUB_RAW_URL}"
fi
